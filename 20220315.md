using namespace std;
int check(vector<int> & nums,int x){
    int l=0,num=0,ls=0;
    for(int i=0;i<nums.size();i++){
        ls=ls+nums[i];
        while(ls>x){
            ls-=nums[l];
            l++;
        }
        num+=i-l+1;
    }
    return num;
}
int fun(vector<int> nums,int k){
    int l=1,r=1e9,mid,ans;
    while(l<=r){
        mid=l+r>>1;
        if(check(nums,mid)>=k){
            ans=mid;
            r=mid-1;
        }else{
            l=mid+1;
        }
    }
    return ans;
}
